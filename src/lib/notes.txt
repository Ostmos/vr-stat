NumericalGrid {
    dimensions: {x, y, z}
    steps: {x, y, z}
    data: {x[], y[], z[]}

    stepLength{x, y, z} = dimension{x, y, z} / steps{x, y, z}
    // 3 planes: xz, yz, xz
    origin = {-d.x/2,-d.y/2,-d.z/2}
    2dGrid(steps{x, y}, stepLength{x, y}) 
    labels(start{x, y, z}, end{x, y, z}, stepLength) 
}
